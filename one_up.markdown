<script type="text/javascript"> 
(function(){
  var initial = "number = int(input(&quot;Please enter a number: &quot;))\n" +
    "number = number + 1\n" +
    "print(number)\n" +
    "number = input(&quot;Please enter a number: &quot;) #distractor\n" +
    "print(&quot;Please enter a number: &quot;) #distractor\n" +
    "input(number) #distractor";
  var parsonsPuzzle = new ParsonsWidget({
    "sortableId": "sortable",
    "max_wrong_lines": 10,
    "grader": ParsonsWidget._graders.LineBasedGrader,
    "exec_limit": 2500,
    "can_indent": true,
    "x_indent": 50,
    "lang": "en",
    "show_feedback": true,
    "trashId": "sortableTrash"
  });
  parsonsPuzzle.init(initial);
  parsonsPuzzle.shuffleLines();
  $("#newInstanceLink").click(function(event){ 
      event.preventDefault(); 
      parsonsPuzzle.shuffleLines(); 
  }); 
  $("#feedbackLink").click(function(event){ 
      event.preventDefault(); 
      parsonsPuzzle.getFeedback(); 
  }); 
})(); 
</script>
